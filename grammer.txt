
expr::= term ({PLUS/MINUS} term)* | 

term: factor ({MUL/Div} factor)* 
factor: INT  #NO floats
      : (PLUS |MINUS ) factor
      : (LPAREN) factor (RPAREN)

<expr> ::= <arith_expr> | <bool_expr>
<bool_expr> ::=  <arith_expr> <comparison> <arith_expr> | <bool_expr> (&& | ||) <bool_expr> | !<bool_expr> | <BOOLEAN>
<comparison> ::= == | != | >= | > | < | <= 
<BOOLEAN> ::= TRUE | FALSE
<arith_expr>: <term> | <arith_expr> (PLUS/MINUS) <term>
<term>: <factor> | <term> (MUL/DIV) <factor> 
<factor>: <INT> 
      : (PLUS|MINUS) factor
      : (LPAREN) factor (RPAREN)
